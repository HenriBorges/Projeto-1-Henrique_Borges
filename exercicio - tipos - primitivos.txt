package fundamentos;

public class TiposPrimitivos {

	public static void main(String[] args) {

		// Tipos númericos inteiros - sem pontos flutuantes
		byte anosDeEmpresa = 23;
		short numeroDeVoos = 542;
		int id = 56789;
		long pontosAcumulados = 3_134_845_223L; // obrigátorio apos 2 bilhoes colocar o L

		// Tipos numéricos reais - com pontos flutuantes
		float salario = 11_445.44F; // para se referir a flout COM PONTO FLUTUANTE tem que por oo F
		double vendasAcumuladas = 2_991_797_103.01;

		// Tipo booleano falso/verdadeiro
		boolean estaDeFerias = false; // true

		// Tipo caractere
		char status = 'A'; // Ativo

		// Dias de empresa
		System.out.println(anosDeEmpresa * 365);

		// nuemros de viagens
		System.out.println(numeroDeVoos / 2);

		// Pontos por real
		System.out.println(pontosAcumulados / vendasAcumuladas);

		System.out.println(id + ":ganha :" + salario);
		System.out.println("Férias? " + estaDeFerias);
		System.out.println("Status : " + status);

	}

}
